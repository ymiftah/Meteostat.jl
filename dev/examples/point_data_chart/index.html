draw figure

fig = (
    AOG.data(weather_data) *
    mapping(
        :time,
        [:temp, :dwpt];
        color=dims(1) => renamer(["Measured", "Dewpoint"]) => "Temperatures",
    ) *
    visual(Lines)
);
img = DisplayAs.PNG(
    draw(fig; figure=(; title="Temperature in Brisbane, QLD"), axis=(; width=400))
)
